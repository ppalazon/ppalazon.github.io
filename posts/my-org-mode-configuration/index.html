Personal org-mode configuration

(require 'package)
(add-to-list 'package-archives '("org" . "https://orgmode.org/elpa/") t)

(require 'org)

;; Enabled ox-hugo
(with-eval-after-load 'ox
  (require 'ox-hugo))

;; Configuration of org-mode
(setq org-startup-indented t)
(setq org-startup-folded "showall")
(setq org-directory "~/org")
(setq org-capture-directory (concat org-directory "/capture"))
(setq org-publish-directory (concat org-directory "/publish"))
(setq org-ref-directory (concat org-directory "/ref"))
(setq org-default-notes-file (concat org-capture-directory "/notes.org"))

;; http://www.howardism.org/Technical/Emacs/journaling-org.html
(setq org-journal-dir (concat org-directory "/journal"))
(defun get-journal-file-today ()
  "Return filename for today's journal entry."
  (let ((daily-name (format-time-string "%Y%m%d")))
    (expand-file-name (concat org-journal-dir (concat "/" daily-name)))))

(defun journal-file-today ()
  "Create and load a journal file based on today's date."
  (interactive)
  (find-file (get-journal-file-today)))

(add-to-list 'auto-mode-alist '(".*/[0-9]*$" . org-mode))

(defun journal-file-insert ()
  "Insert's the journal heading based on the file's name."
  (interactive)
  (when (string-match "\\(20[0-9][0-9]\\)\\([0-9][0-9]\\)\\([0-9][0-9]\\)"
                      (buffer-name))
    (let ((year (string-to-number (match-string 1 (buffer-name))))
          (month (string-to-number (match-string 2 (buffer-name))))
          (day (string-to-number (match-string 3 (buffer-name))))
          (datim nil))
      (setq datim (encode-time 0 0 0 day month year))
      (insert (format-time-string
               "#+TITLE: Journal Entry - %Y-%b-%d (%A)\n\n" datim))))
  )

(require 'autoinsert)
(add-hook 'find-file-hook 'auto-insert)
(add-to-list 'auto-insert-alist '(".*/journal/[0-9]*$" . journal-file-insert))

(setq org-agenda-files (list org-capture-directory org-journal-dir org-ref-directory))
(setq org-agenda-files-regexp "\\`[^.].*\\.org'\\|[0-9]+$")
(setq org-capture-templates
      '(("t" "Todo" entry (file+headline (concat org-capture-directory "/gtd.org") "Tasks")
         "* TODO %?\n  %i\n  %a")
        ("i" "Idea" entry (file+headline (concat org-capture-directory "/notes.org") "Ideas")
         "* %?\n  %i"
         :empty-lines 1)
        ("o" "Order" entry (file+headline (concat org-capture-directory "/notes.org") "Orders")
         "* %?\n  %i"
         :empty-lines 1)
        ("b" "Blog Idea" entry (file+headline (concat org-capture-directory "/notes.org") "Blog Ideas")
         "* %?\n  %i"
         :empty-lines 1)
        ("j" "Journal" entry (file (get-journal-file-today))
         "* %<%H:%M>: %?\nEntered on %U\n%i\nFrom: %a"
         :empty-lines 1)))
(setq org-agenda-include-diary t)
(setq org-publish-project-alist
      '(("personal-blog"
         :base-directory "~/org/personal-blog"
         :base-extension "org"
         :publishing-directory "~/org/publish/personal-blog"
         :recursive t
         :publishing-function org-html-publish-to-html
         :headline-levels 4
         :html-extension "html"
         :body-only t)))

(setq org-latex-pdf-process
      '("xelatex -interaction nonstopmode %f"
        "xelatex -interaction nonstopmode %f"))

(setq org-latex-logfiles-extensions
      (quote ("lof" "lot" "tex" "tex~" "aux" "idx"
              "log" "out" "toc" "nav" "snm" "vrb"
              "dvi" "fdb_latexmk" "blg" "brf" "fls"
              "entoc" "ps" "spl" "bbl")))

(setq org-src-fontify-natively t)
(setq org-export-coding-system 'utf-8)

(require 'org-download)
(setq-default org-download-image-dir (concat org-directory "/pictures"))
(setq org-image-actual-width nil)
